# IMPORTANT NOTES FOR LINUX/WSL2 GPU SETUP:
#
# 1. ENSURE NVIDIA DRIVER IS INSTALLED:
#    - Check with: nvidia-smi
#    - Install from: https://www.nvidia.com/en-us/geforce/drivers/
#
# 2. ENSURE CUDA TOOLKIT IS INSTALLED:
#    - TensorFlow 2.15+ requires CUDA 12.3+
#    - Download from: https://developer.nvidia.com/cuda-downloads
#    - Ubuntu WSL2 example:
#      wget https://developer.download.nvidia.com/compute/cuda/repos/wsl-ubuntu/x86_64/cuda-wsl-ubuntu.pin
#      sudo mv cuda-wsl-ubuntu.pin /etc/apt/preferences.d/cuda-repository-pin-wsl-ubuntu
#      wget https://developer.download.nvidia.com/compute/cuda/repos/wsl-ubuntu/x86_64/cuda-repo-wsl-ubuntu-12-4-local_12.4.0-1_amd64.deb
#      sudo dpkg -i cuda-repo-wsl-ubuntu-12-4-local_12.4.0-1_amd64.deb
#      sudo apt-key adv --fetch-keys https://developer.download.nvidia.com/compute/cuda/repos/wsl-ubuntu/x86_64/3bf863cc.pub
#      sudo apt-get update
#      sudo apt-get -y install cuda
#
# 3. ENSURE cuDNN IS INSTALLED:
#    - Download from: https://developer.nvidia.com/cudnn
#    - Should be at /usr/local/cuda/include/cudnn.h
#
# 4. SET ENVIRONMENT VARIABLES:
#    Add to ~/.bashrc or ~/.zshrc:
#    export CUDA_HOME=/usr/local/cuda
#    export PATH=$CUDA_HOME/bin:$PATH
#    export LD_LIBRARY_PATH=$CUDA_HOME/lib64:$LD_LIBRARY_PATH
#
# 5. VERIFY GPU SETUP:
#    python gpu_setup_verification.py
#
# 6. IF USING WSL2:
#    WSL2 already supports GPU passthrough via NVIDIA CUDA on WSL2
#    No special configuration needed - just install CUDA inside WSL2
